368 {
369  bool err(false);
370  XM_ENSURE_TRUE(a_outPoly.size() > 1, true);
371 
372  // the angles at the corners must be convex
373  VecInt c;
374  c.reserve(a_polyCorners.size() + 1);
375  c.push_back(0);
376  for (size_t i = 0; i < a_polyCorners.size(); ++i)
377  c.push_back(a_polyCorners[i]);
378 
379  Pt3d p0, p1, p2;
380  p0 = a_outPoly.back();
381  p1 = a_outPoly.front();
382  p2 = a_outPoly[1];
383  double angle = gmAngleBetweenEdges(p0, p1, p2);
384  if (angle >= XM_PI)
385  err = true;
386  for (size_t i = 1; i < c.size() && !err; ++i)
387  {
388  if (c[i] == c[i - 1])
389  continue; // skip when we have a merged node
390  if (c[i] - 1 < 0)
391  {
392  std::string msg = "Invalid corner detected in polygon. Aborting patch.";
393  meModifyMessageWithPolygonId(m_polyId, msg);
394  XM_LOG(xmlog::error, msg);
395  return true;
396  }
397  p0 = a_outPoly[c[i] - 1];
398  p1 = a_outPoly[c[i]];
399  if (c[i] + 1 < (int)a_outPoly.size())
400  p2 = a_outPoly[c[i] + 1];
401  else
402  p2 = a_outPoly[0];
403  angle = gmAngleBetweenEdges(p0, p1, p2);
404  if (angle >= XM_PI)
405  err = true;
406  }
407 
408  if (err)
409  {
410  std::string msg = "Non-convex corner detected in polygon. Aborting patch.";
411  meModifyMessageWithPolygonId(m_polyId, msg);
412  XM_LOG(xmlog::error, msg);
413  return true;
414  }
415  // Check number of sides. There should be 3 or 4. If there are 3 we treat
416  // the corner with the smallest angle as the "merged node."
417  return false;

784 //------------------------------------------------------------------------------
785 void MePolyPatcherImpl::SetUpColumnForRectPatch3a(int a_j, VecPt3d& a_newcol)
786 {
787  a_newcol.resize(m_np_end, Pt3d());
788  int np = m_nodesincol[a_j];
789  int np_side = m_nodesinrow.back();
790  // get corners
791  Pt3d c[4];
792  c[0] = m_end1pts[a_j][0];
793  c[1] = m_end1pts[a_j][np - 1];
794  c[2] = m_end2pts[a_j][0];
795  c[3] = m_end2pts[a_j][np - 1];
796  // find points on new column
797  for (int i = 1; i < np - 1; ++i)
798  {
799  double u_pcnt = (double)a_j / (np_side - 1);
800  int row = (int)((double)i / (np - 1) * (m_np_end - 1) + 0.5);
801  int col = (int)(u_pcnt * (m_nodesinrow[row] - 1) + 0.5);
802  // avoid messing up the mesh, find least row that gives same row
803  for (int i_test = i; row > 0 && i_test == i;)
804  {
805  i_test = (int)((double)(row - 1) / (m_np_end - 1) * (np - 1) + 0.5);
806  if (i_test == i)
807  {
808  row--;
809  col = (int)(u_pcnt * (m_nodesinrow[row] - 1) + 0.5);
810  }
811  }
812  // don't create extra nodes on boundary
813  if (col == m_nodesinrow[row] - 1 && col == 1)
814  {
815  a_newcol[i].x = a_newcol[i].y = a_newcol[i].z = -1234.5;
816  }
817  else
818  {
819  // don't create extra nodes on boundary
820  if (col == 0)
821  col++;
822  else if (col == m_nodesinrow[row] - 1)
823  col--;
824  a_newcol[0] = m_side1pts[row][col];
825  a_newcol[np - 1] = m_side2pts[row][col];
826  // avoid messing up the mesh, find least j that gives same column
827  int j_test, col_test;
828  for (j_test = a_j, col_test = col; j_test > 1 && col_test == col;)
829  {
830  col_test =
831  (int)((double)(j_test - 1) / (m_nodesinrow[m_np_end - 1] - 1) * (m_nodesinrow[row] - 1) +
832  0.5);
833  if (col_test == 0)
834  col_test++;
835  else if (col_test == m_nodesinrow[row] - 1)
836  col_test--;
837  if (col_test == col)
838  j_test--;
839  }
840  // find correct i value
841  int i_test = (int)((double)i / (m_nodesincol[a_j] - 1) * (m_nodesincol[j_test] - 1) + 0.5);
842  // handle degenerate case with this if statement
843  if (m_nodesincol[j_test] > 2)
844  {
845  if (i_test == 0)
846  i_test++;
847  else if (i_test == m_nodesincol[j_test] - 1)
848  i_test--;
849  }
850  // compute these
851  int j_min = Mmin(j_test, m_nodesinrow[i_test] - 1);
852  u_pcnt = (double)j_min / (double)(m_nodesinrow[i_test] - 1); // j_min
853  double v_pcnt = (double)i_test / (double)(m_nodesincol[j_test] - 1); // j_test
854  // now get the correct percentages
855  double u1 =
856  m_newside1pcnt[i_test][j_min] * (1.0 - v_pcnt) + m_newside2pcnt[i_test][j_min] * v_pcnt;
857  double u2 = 1.0 - u1;
858  double v1 =
859  m_newend1pcnt[j_test][i_test] * (1.0 - u_pcnt) + m_newend2pcnt[j_test][i_test] * u_pcnt;
860  double v2 = 1.0 - v1;
861  // finally, compute the x, y, z components from the Coons equation
862  a_newcol[i].x = u2 * m_end1pts[j_test][i_test].x + u1 * m_end2pts[j_test][i_test].x +
863  v2 * a_newcol[0].x + v1 * a_newcol[np - 1].x -
864  u2 * (v2 * c[0].x + v1 * c[1].x) - u1 * (v2 * c[2].x + v1 * c[3].x);
865  a_newcol[i].y = u2 * m_end1pts[j_test][i_test].y + u1 * m_end2pts[j_test][i_test].y +
866  v2 * a_newcol[0].y + v1 * a_newcol[np - 1].y -
867  u2 * (v2 * c[0].y + v1 * c[1].y) - u1 * (v2 * c[2].y + v1 * c[3].y);
868  a_newcol[i].z = u2 * m_end1pts[j_test][i_test].z + u1 * m_end2pts[j_test][i_test].z +
869  v2 * a_newcol[0].z + v1 * a_newcol[np - 1].z -
870  u2 * (v2 * c[0].z + v1 * c[1].z) - u1 * (v2 * c[2].z + v1 * c[3].z);
871  }
872  }

877 void MePolyPatcherImpl::GenerateMeshCells()
878 {
879  // These correspond to defines in vtkCellType. There are classes downstream from
880  // meshing that convert a cell stream into a vtkUnstructured grid that use these
881  // magic numbers.
882  const int VTK_QUAD(9);
883  const int VTK_TRI(5);
884 
885  // loop through columns
886  for (int j = 1; j < m_np_side; ++j)
887  {
888  // find nodes for next element as follows
889  //------------------------------------------------------------------------------
890  // node1a node2a
891  // *-------*
892  // | |
893  // | |
894  // | |
895  // *-------*
896  // node1b node2b
897  //------------------------------------------------------------------------------
898  int i(0), index1(0), index2(0);
899  for (i = 1, index1 = 0; i < m_nodesincol[j]; ++i)
900  { // find index2
901  if (m_nodesincol[j] == m_nodesincol[j - 1])
902  index2 = i;
903  else
904  {
905  double fval = (double)i / (m_nodesincol[j] - 1);
906  if (fval >= 0.50)
907  index2 = (int)(fval * (m_nodesincol[j - 1] - 1) + 0.51);
908  else
909  index2 = (int)(fval * (m_nodesincol[j - 1] - 1) + 0.49);
910  // make sure index2 is okay
911  index2 = Mmin(index2, index1 + 1);
912  }
913  // get nodes for new cell
914  int node1a(m_nodeIdx[index1][j - 1]), node1b(m_nodeIdx[index2][j - 1]),
915  node2a(m_nodeIdx[i - 1][j]), node2b(m_nodeIdx[i][j]);
916  index1 = index2;
917  // see if the element got pinched to a line
918  if ((node1a == node1b && node2a == node2b) || (node1a == node2a && node1b == node2b) ||
919  (node1a == node1b && node1a == node2a) || (node1b == node1a && node1b == node2b) ||
920  (node2b == node1b && node2b == node2a) || (node2a == node1a && node2a == node2b))
921  { // do not create element in this case
922  }
923  else
924  { // make sure nodes are ccw
925  int x[4] = {node1a, node1b, node2b, node2a};
926  VecInt n_ccw(&x[0], &x[4]);
927  VecInt idxs(4);
928  VecPt3d pts(4, Pt3d());
929  for (int i = 0; i < 4; ++i)
930  {
931  idxs[i] = i;
932  pts[i] = (*m_meshPts)[n_ccw[i]];
933  }
934  gmOrderPointsCounterclockwise(pts, idxs);
935  node1a = n_ccw[idxs[0]];
936  node1b = n_ccw[idxs[1]];
937  node2b = n_ccw[idxs[2]];
938  node2a = n_ccw[idxs[3]];
939  // see if new quad will be built
940  if (node1a != node1b && node2a != node2b && node1a != node2a && node1b != node2b)
941  {
942  m_meshCells.push_back(VTK_QUAD); // cell type
943  m_meshCells.push_back(4); // number of points
944  m_meshCells.push_back(node1a); // the points that make up the
945  m_meshCells.push_back(node1b); // cell
946  m_meshCells.push_back(node2b);
947  m_meshCells.push_back(node2a);
948  }
949  else
950  { // triangle will be built -- find which one
951  int ix[3];
952  ix[0] = node1a;
953  if (node1a == node1b)
954  {
955  ix[1] = node2b;
956  ix[2] = node2a;
957  }
958  else if (node2a == node2b)
959  {
960  ix[1] = node1b;
961  ix[2] = node2a;
962  }
963  else if (node1a == node2a)
964  {
965  ix[1] = node1b;
966  ix[2] = node2b;
967  }
968  else
969  {
970  ix[1] = node1b;
971  ix[2] = node2a;
972  }
973  m_meshCells.push_back(VTK_TRI);
974  m_meshCells.push_back(3);
975  for (int i = 0; i < 3; ++i)
976  m_meshCells.push_back(ix[i]);
977  }
978  }
979  }
980  }

1020 //------------------------------------------------------------------------------
1021 bool MePolyPatcherImpl::CellOverlapsAdj(int a_cellIdx, DynBitset& a_cellFlags)
1022 {
1023  const VecPt3d& mp(*m_meshPts);
1024  VecInt cellPtIdx, adjCellIdx;
1025  VecInt2d adjPtIdx;
1026  GetCellPtInfo(a_cellIdx, cellPtIdx, adjCellIdx, adjPtIdx);
1027  VecPt3d cellPoints;
1028  Pt3d cellCenter;
1029  for (size_t i = 0; i < cellPtIdx.size(); ++i)
1030  {
1031  cellPoints.push_back(mp[cellPtIdx[i]]);
1032  cellCenter += cellPoints.back();
1033  }
1034  cellCenter /= cellPoints.size();
1035 
1036  // add the first idx to the end of the vector so that we can test the edge defined by
1037  // the last point to the first point
1038  cellPtIdx.push_back(cellPtIdx.front());
1039  // check each edge to see if it overlaps the edges of any neighbor cell
1040  for (size_t i = 1; i < cellPtIdx.size(); ++i)
1041  {
1042  int i0(cellPtIdx[i - 1]), i1(cellPtIdx[i]);
1043  const Pt3d &p0(mp[i0]), &p1(mp[i1]);
1044  int cnt(-1);
1045  for (auto v : adjPtIdx)
1046  {
1047  cnt++;
1048  if (a_cellFlags[adjCellIdx[cnt]])
1049  continue;
1050  Pt3d adjCellCenter;
1051  VecPt3d adjacentPoints;
1052  for (size_t j = 0; j < v.size(); ++j)
1053  {
1054  adjacentPoints.push_back(mp[v[j]]);
1055  adjCellCenter += adjacentPoints.back();
1056  }
1057  adjCellCenter /= adjacentPoints.size();
1058 
1059  bool overlap(false);
1060  // see if centroid of a_cellIdx is inside of neigh
1061  if (gmPointInPolygon2D(&adjacentPoints[0], adjacentPoints.size(), cellCenter) > -1)
1062  {
1063  std::string msg = "Invalid patch. Centroid of base cell inside of adjacent cell.";
1064  meModifyMessageWithPolygonId(m_polyId, msg);
1065  XM_LOG(xmlog::error, msg);
1066  return true;
1067  }
1068 
1069  // see if centroid of neigh is inside of a_cellIdx
1070  if (gmPointInPolygon2D(&cellPoints[0], cellPoints.size(), adjCellCenter) > -1)
1071  {
1072  std::string msg = "Invalid patch. Centroid of adjacent cell inside of base cell.";
1073  meModifyMessageWithPolygonId(m_polyId, msg);
1074  XM_LOG(xmlog::error, msg);
1075  return true;
1076  }
1077 
1078  // add the first idx to the end of the vector so that we can test the edge defined by
1079  // the last point to the first point
1080  v.push_back(v.front());
1081  for (size_t j = 1; !overlap && j < v.size(); ++j)
1082  {
1083  int j0(v[j - 1]), j1(v[j]);
1084  // edges share a point in the grid so don't check if they intersect
1085  if (i0 == j0 || i0 == j1 || i1 == j0 || i1 == j1)
1086  continue;
1087 
1088  const Pt3d &p2(mp[j0]), &p3(mp[j1]);
1089  if (gmLinesIntersect(p0, p1, p2, p3))
1090  {
1091  std::string msg = "Invalid patch. Edges of adjacent cells overlap.";
1092  meModifyMessageWithPolygonId(m_polyId, msg);
1093  XM_LOG(xmlog::error, msg);
1094  return true;
1095  }
1096  }
1097  }
1098  }
1099  a_cellFlags[a_cellIdx] = true;
1100  return false;
1101 } // MePolyPatcherImpl::CellOverlapsAdj
