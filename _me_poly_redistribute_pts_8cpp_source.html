648  finalpts[numfinal - 1].w = totdist;
649  len_segs += totdist;
650  prevpt = finalpts[numfinal - 1];
651  totdist = 0.0;
652  // get length of segment left after inserting pt
653  length -= ptlength;
654  pcntin = 0.0;
655  }
656  else
657  { // go on to next segment
658  pcntin = pcntleft;
659  if (++nextid < numint)
660  {
661  prevpt = nextpt;
662 
663  nextpt = a_pts[nextid];
664  nextpt.z = a_interpLengths[nextid];
665  totdist += length;
666  length = Mdist(prevpt.x, prevpt.y, nextpt.x, nextpt.y);
667  len_arc += length;
668  }
669  else
670  done = true;
671  }
672  }
673 
674  // make sure we still have an arc
675  if (numfinal > 0)
676  {
677  double len_segs_to_adjust = 0.0;
678  // init length of the arc to adjust to remaining portion
679  double len_arc_to_adjust = len_arc - len_segs;
680  // init first point to move as the last in the list
681  int first_pt_to_move = numfinal - 1;
682  // eliminate last part_seg if less than half
683  if (pcntin < 0.5)
684  --numfinal;
685  // keep the length of the last seg if more than half
686  else
687  len_segs_to_adjust = (length + totdist) / pcntleft;
688  // find segs with length close to final length to find first pt to move
689  while (first_pt_to_move >= 0 && finalpts[first_pt_to_move].w < w1 * 2.0 &&
690  finalpts[first_pt_to_move].w > w1 * 0.5)
691  {
692  len_segs_to_adjust += finalpts[first_pt_to_move].w;
693  len_arc_to_adjust += finalpts[first_pt_to_move].w;
694  --first_pt_to_move;
695  }
696  ++first_pt_to_move;
697  // find error adjustment scale
698  double scale = len_arc_to_adjust / len_segs_to_adjust;
699  /* reset the locations using scaled distance */
700  nextid = 1;
701  prevpt = a_pts[0];
702  prevpt.z = a_interpLengths[0];
703  nextpt = a_pts[1];
704  nextpt.z = a_interpLengths[1];
705  length = Mdist(prevpt.x, prevpt.y, nextpt.x, nextpt.y);
706  for (int i = 0; i < numfinal; i++)
707  {
708  if (i < first_pt_to_move)
709  {
710  totdist = finalpts[i].w;
711  // move along int points until reach finalpt
712  while (totdist > length)
713  {
714  totdist -= length;
715  // go onto next segment
716  prevpt = nextpt;
717  ++nextid;
718  nextpt = a_pts[nextid];
719  nextpt.z = a_interpLengths[nextid];
720  length = Mdist(prevpt.x, prevpt.y, nextpt.x, nextpt.y);
721  }
722  }
723  else
724  {
725  totdist = finalpts[i].w * scale;
726  // move along int points until reach finalpt
727  while (totdist > length)
728  {
729  totdist -= length;
730  // go onto next segment
731  prevpt = nextpt;
732  ++nextid;
733  nextpt = a_pts[nextid];
734  nextpt.z = a_interpLengths[nextid];
735  length = Mdist(prevpt.x, prevpt.y, nextpt.x, nextpt.y);
736  }
737  // reached point to move
738  double pcnt = totdist / length;
739  finalpts[i].x = prevpt.x + pcnt * (nextpt.x - prevpt.x);
740  finalpts[i].y = prevpt.y + pcnt * (nextpt.y - prevpt.y);
741  }
742  prevpt = finalpts[i];
743  length -= totdist;
744  }
745  }
746 
747  VecPt3d ret;
748  // if arc closes, make sure there are at least 3 points on arc
749  if (a_pts.front() == a_pts.back() && numfinal < 2)
750  {
751  if (a_pts.size() < 3)
752  return ret;
753 
754  finalpts.resize(2);
755  finalpts[0] = a_pts[a_pts.size() / 3];
756  finalpts[1] = a_pts[a_pts.size() * 2 / 3];
757  }
758  ret.assign(finalpts.size() + 2, Pt3d());
759  ret.front() = a_pts.front();
760  ret.back() = a_pts.back();
761  for (size_t i = 1; i < finalpts.size(); ++i)
762  ret[i] = finalpts[i];
763  return ret;
764 } // MePolyRedistributePtsImpl::RedistPts2

802 void MePolyRedistributePtsImpl::TvaluesForSeg(size_t a_idx,
803  size_t a_nextIdx,
804  const VecDbl& a_segLengths,
805  const VecDbl& a_segTvalues,
806  const VecDbl& a_interpLengths,
807  double& a_pcntin,
808  VecDbl& a_tVals)
809 {
810  // current segment length
811  double length(a_segLengths[a_idx]);
812  // tvalues at the begin and end of current segment
813  double t1(a_segTvalues[a_idx]), t2(a_segTvalues[a_nextIdx]);
814  if (0 == a_nextIdx)
815  t2 = 1;
816  // target segment length at begin and end of segment
817  double w1(a_interpLengths[a_idx]), w2(a_interpLengths[a_nextIdx]);
818 
819  bool done(false);
820  while (!done)
821  {
822  double pcntleft = a_pcntin + (2 * length) / (w1 + w2);
823  if (pcntleft >= 1.0)
824  {
825  double scale = 1.0 - a_pcntin;
826  double ptlength = 2 * scale * w1 * length / (2 * length + scale * (w1 - w2));
827  double pcnt = ptlength / length;
828  double t = (pcnt * (t2 - t1)) + t1; // convert to polyline tval
829  a_tVals.push_back(t);
830 
831  w1 = w1 * (1 - pcnt) + w2 * pcnt;
832  t1 = t;
833  length -= ptlength;
834  a_pcntin = 0;
835  }
836  else
837  {
838  a_pcntin = pcntleft;
839  done = true;
840  }
841  }
842 } // MePolyRedistributePtsImpl::TvaluesForSeg

863 void MePolyRedistributePtsImpl::SizeFromPolyAddEdgeLengths(const VecPt3d& a_pts)
864 {
865  double l(0), dx, dy;
866  size_t start = m_polyEdgeLengths.size();
867  m_polyEdgeLengths.insert(m_polyEdgeLengths.end(), a_pts.size(), 0);
868  for (size_t i = 0; i < a_pts.size(); ++i)
869  {
870  size_t nextIdx(0);
871  Pt3d nextPt(a_pts[0]);
872  if (i + 1 < a_pts.size())
873  {
874  nextIdx = i + 1;
875  nextPt = a_pts[i + 1];
876  }
877  m_polyPts->push_back(a_pts[i]);
878 
879  dx = a_pts[i].x - nextPt.x;
880  dy = a_pts[i].y - nextPt.y;
881  l = sqrt(dx * dx + dy * dy);
882  if (l < m_minLength)
883  m_minLength = l;
884  if (l > m_maxLength)
885  m_maxLength = l;
886  l = l / 2;
887  m_polyEdgeLengths[i + start] += l;
888  m_polyEdgeLengths[nextIdx + start] += l;
889  }
890 } // MePolyRedistributePtsImpl::SizeFromPolyAddEdgeLengths
