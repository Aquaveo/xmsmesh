381 void MePolyPaverToMeshPtsImpl::ClassifyPolys()
382 {
383  if (m_offsetOutputs[0].m_pts.empty())
384  return;
385 
386  std::vector<std::vector<size_t>> polys;
387  MeIntersectPolys ip;
388  ip.ClassifyPolys(m_offsetOutputs[0], polys);
389 
390  // put these on the stack
391  std::vector<Pt3d>& pts(m_offsetOutputs[0].m_pts);
392  Poly p;
393  p.m_iter = m_polyOffsetIter + 1;
394  for (size_t i = 0; i < polys.size(); ++i)
395  {
396  p.m_outside.resize(0);
397  p.m_inside.resize(0);
398  // get the outside loop
399  std::vector<size_t>& oLoop(m_offsetOutputs[0].m_loops[polys[i][0]]);
400  for (size_t j = 0; j < oLoop.size(); ++j)
401  {
402  p.m_outside.push_back(pts[oLoop[j]]);
403  }
404  // do inside loops
405  for (size_t j = 1; j < polys[i].size(); ++j)
406  {
407  std::vector<size_t>& iLoop(m_offsetOutputs[0].m_loops[polys[i][j]]);
408  p.m_inside.push_back(std::vector<Pt3d>());
409  for (size_t k = 0; k < iLoop.size(); ++k)
410  {
411  p.m_inside.back().push_back(pts[iLoop[k]]);
412  }
413  }
414 
415  m_polyStack.push_back(p);
416  }

422 static double iEnvelopeArea(const std::vector<Pt3d>& a_pts)
423 {
424  double area;
425  Pt3d pMin(XM_DBL_HIGHEST),
426  pMax(XM_DBL_LOWEST); // changed to XM_DBL_LOWEST because it is an arbitrarily low value
427  for (size_t i = 0; i < a_pts.size(); ++i)
428  {
429  gmAddToExtents(a_pts[i], pMin, pMax);
430  }
431  area = (pMax.x - pMin.x) * (pMax.y - pMin.y);
432  return area;
433 } // iEnvelopeArea
