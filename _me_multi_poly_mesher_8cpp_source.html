373 void MeMultiPolyMesherImpl::CheckForIntersections(const MeMultiPolyMesherIo& a_io,
374  std::string& a_errors) const
375 {
376  // Create vectors defining all poly segments and where they're located
377 
378  std::vector<const Pt3d*> segments;
379  std::vector<SegmentLocation> segmentLocs;
380  for (int i = 0; i < (int)a_io.m_polys.size(); ++i)
381  {
382  if (i > 0)
383  {
384  // Add a null to separate polygons
385  segments.push_back(nullptr);
386  segmentLocs.push_back(SegmentLocation());
387  }
388 
389  const MePolyInput& inputPoly = a_io.m_polys[i];
390  const VecPt3d& outerPoly = inputPoly.m_outPoly;
391 
392  // Do the outer poly
393  for (int j = 0; j < (int)outerPoly.size(); ++j)
394  {
395  segments.push_back(&outerPoly[j]);
396  segmentLocs.push_back(SegmentLocation(i, j, -1, -1));
397  }
398  // Do the last segment because polygon isn't closed
399  segments.push_back(&outerPoly[0]);
400  segmentLocs.push_back(SegmentLocation(i, 0, -1, -1));
401 
402  // Do the inner polys
403 
404  if (!inputPoly.m_insidePolys.empty())
405  {
406  // Add a null to separate polygons
407  segments.push_back(nullptr);
408  segmentLocs.push_back(SegmentLocation());
409  }
410 
411  for (int k = 0; k < (int)inputPoly.m_insidePolys.size(); ++k)
412  {
413  if (k > 0)
414  {
415  // Add a null to separate polygons
416  segments.push_back(nullptr);
417  segmentLocs.push_back(SegmentLocation());
418  }
419  const VecPt3d& innerPoly = inputPoly.m_insidePolys[k];
420 
421  // Do the inner poly
422  for (int m = 0; m < (int)innerPoly.size(); ++m)
423  {
424  segments.push_back(&innerPoly[m]);
425  segmentLocs.push_back(SegmentLocation(i, -1, k, m));
426  }
427  // Do the last segment because polygon isn't closed
428  segments.push_back(&innerPoly[0]);
429  segmentLocs.push_back(SegmentLocation(i, -1, k, 0));
430  }
431  }
432 
433  // Intersect all segments with all other segments
434 
435  double xi, yi, zi1, zi2, tol(gmXyTol());
436  size_t size = segments.size();
437 
438  // Loop through every segment
439  for (size_t i = 1; i < segments.size(); ++i)
440  {
441  // Skip nulls which indicate the end of a poly
442  if (segments[i] == nullptr)
443  {
444  ++i;
445  continue;
446  }
447  const Pt3d* one1 = segments[i - 1];
448  const Pt3d* one2 = segments[i];
449  Pt3d oneMn, oneMx;
450  gmAddToExtents(*one1, oneMn, oneMx);
451  gmAddToExtents(*one2, oneMn, oneMx);
452  oneMn -= tol;
453  oneMx += tol;
454 
455  // Starting on the next segment from where we are, go to end of segments
456  for (size_t j = i + 1; j < size; ++j)
457  {
458  // Skip nulls which indicate the end of a poly
459  if (segments[j] == nullptr)
460  {
461  ++j;
462  continue;
463  }
464  const Pt3d* two1 = segments[j - 1];
465  const Pt3d* two2 = segments[j];
466  if (ExtentsOverlap(oneMn, oneMx, *two1, *two2))
467  {
468  if (gmIntersectLineSegmentsWithTol(*one1, *one2, *two1, *two2, &xi, &yi, &zi1, &zi2, tol))
469  {
470  // See if we didn't just intersect on the ends
471  if (!gmEqualPointsXY(one1->x, one1->y, xi, yi, tol) &&
472  !gmEqualPointsXY(one2->x, one2->y, xi, yi, tol))
473  {
474  // Report the intersection
475  std::stringstream ss;
476  SegmentLocation& a0 = segmentLocs[i - 1];
477  SegmentLocation& a1 = segmentLocs[i];
478  SegmentLocation& b0 = segmentLocs[j - 1];
479  SegmentLocation& b1 = segmentLocs[j];
480  std::string s;
481  if (a0.m_inPoly == -1)
482  { // First segment on an outer poly
483  if (b0.m_inPoly == -1)
484  { // Second segment on an outer poly
485  s = (boost::format("Error: Input polygon segments intersect."
486  " The segment defined by points %d and %d"
487  " of outer polygon %d"
488  " intersects with the segment defined by points %d and %d"
489  " of outer polygon %d.\n") %
490  a0.m_polySeg % a1.m_polySeg % a0.m_poly % b0.m_polySeg % b1.m_polySeg %
491  b0.m_poly)
492  .str();
493  }
494  else
495  { // Second segment on an inner poly
496  s = (boost::format("Error: Input polygon segments intersect."
497  " The segment defined by points %d and %d"
498  " of outer polygon %d"
499  " intersects with the segment defined by points %d and %d"
500  " of inner polygon %d of outer polygon %d.\n") %
501  a0.m_polySeg % a1.m_polySeg % a0.m_poly % b0.m_inPolySeg % b1.m_inPolySeg %
502  b0.m_inPoly % b0.m_poly)
503  .str();
504  }
505  }
506  else
507  { // First segment on an inner poly
508  if (b0.m_inPoly == -1)
509  { // Second segment on an outer poly
510  s = (boost::format("Error: Input polygon segments intersect."
511  " The segment defined by points %d and %d"
512  " of inner polygon %d of outer polygon %d"
513  " intersects with the segment defined by points %d and %d"
514  " of outer polygon %d.\n") %
515  a0.m_inPolySeg % a1.m_inPolySeg % a0.m_inPoly % a0.m_poly % b0.m_polySeg %
516  b1.m_polySeg % b0.m_poly)
517  .str();
518  }
519  else
520  { // Second segment on an inner poly
521  s = (boost::format("Error: Input polygon segments intersect."
522  " The segment defined by points %d and %d"
523  " of inner polygon %d of outer polygon %d"
524  " intersects with the segment defined by points %d and %d"
525  " of inner polygon %d of outer polygon %d.\n") %
526  a0.m_inPolySeg % a1.m_inPolySeg % a0.m_inPoly % a0.m_poly % b0.m_inPolySeg %
527  b1.m_inPolySeg % b0.m_inPoly % b0.m_poly)
528  .str();
529  }
530  }
531  a_errors += s;
532  }
533  }
534  }
535  }
536  }

583 {

586  // magic numbers.
587  const int VTK_TRI(5);
588 
589  VecInt cells;
590  cells.swap(a_cells);
591  int cellCount = 0;
592  if (!a_triangles.empty())
593  {
594  cellCount = (int)a_triangles.size() / 3;
595  cells.reserve(cellCount * 5);
596  for (size_t i = 0; i < a_triangles.size(); i += 3)
597  {
598  cells.push_back(VTK_TRI);
599  cells.push_back(3);
600  cells.push_back(a_triangles[i + 0]);
601  cells.push_back(a_triangles[i + 1]);
602  cells.push_back(a_triangles[i + 2]);
603  }
604  }
605  else
606  {
607  size_t i = 0;
608  while (i < cells.size())
609  {
610  cellCount++;
611  i++; // celltype = m_cells[i++];
612  int npts = cells[i++];
613  i += npts;
614  }
615  }
616  m_cellCount += cellCount;
617 
618  if ((*m_pts).empty())
619  {
620  std::pair<std::pair<double, double>, int> pd;
621  for (size_t i = 0; i < a_points.size(); ++i)
622  {
623  pd.first.first = a_points[i].x;
624  pd.first.second = a_points[i].y;
625  pd.second = (int)i;
626  m_ptHash.insert(pd);
627  }
628  // First time, just add points to m_pts and return
629  m_pts->swap(a_points);
630  m_cells.swap(cells);
631  return;
632  }
633 
634  VecInt oldDups, newDups;
635  int oldNumPts = (int)(*m_pts).size();
636  AddUniquePoints(a_points, oldDups, newDups);
637 
638  RenumberNewMesh(oldNumPts, a_points.size(), oldDups, newDups, cells);
639  AppendNewCells(cells);
640 } // MeMultiPolyMesherImpl::AppendMesh

702  std::iota(newPtsIdx.begin(), newPtsIdx.end(), a_oldNumPts);
703  }
704  else
705  {
706  // Set the new points that are duplicates equal to the old point indices
707  for (size_t i = 0; i < a_newDups.size(); ++i)
708  {
709  newPtsIdx[a_newDups[i]] = a_oldDups[i];
710  }
711 
712  // Renumber, skipping duplicate points we just assigned
713  int cnt(a_oldNumPts);
714  for (size_t i = 0; i < newPtsIdx.size(); ++i)
715  {
716  if (newPtsIdx[i] < 0)
717  {
718  newPtsIdx[i] = cnt;
719  ++cnt;
720  }
721  }
722  }
723 
724  // Fix the cell numbers to correspond to the new point numbers
725  for (size_t i = 1; i < a_cells.size(); ++i)
726  {
727  int nPts = a_cells[i];
728  for (int j = 1; j <= nPts; ++j)
729  {
730  a_cells[i + j] = newPtsIdx[a_cells[i + j]];
731  }
732  i += (1 + nPts);
733  }
734 } // MeMultiPolyMesherImpl::RenumberNewMesh
