829 //------------------------------------------------------------------------------
830 VecEdge MeQuadBlossomImpl::GetInteriorEdges(int a_p, const VecCellData& chain)
831 {
832  VecEdge edges;
833  XM_ASSERT(chain.size() > 0);
834  for (size_t i = 0; i < chain.size() - 1; ++i)
835  {
836  const CellData& last = chain[i];
837  const CellData& face = chain[i + 1];
838  int p1 = face.m_priorPoint;
839  // only create an edge if p1 > a_p. Let the chain of p1 create it otherwise.
840  if (p1 > a_p)
841  {
842  // last: [m_priorPoint=2, m_nextPoint=5, face=3],
843  // current: [m_priorPoint=5, m_nextPoint=8, face=6],
844  // a_p: 4=> [p0=4, p1=5, fL=3, fR=6, pL=2, pR=8]
845 
846  // last: [m_priorPoint=3, m_nextPoint=0, face=0],
847  // current: [m_priorPoint=0, m_nextPoint=1, face=1], a_p: 4
848  // => [p0=4, p1=0, fL=0, fR=1, pL=3, pR=1]
849  edges.push_back({a_p, p1, last.m_cell, face.m_cell, last.m_priorPoint, face.m_nextPoint});
850  }
851  }

864 //------------------------------------------------------------------------------
865 Edge MeQuadBlossomImpl::GetBoundaryEdges(int a_p,
866  const VecCellData& chain,
867  int cost,
868  VecMeEdge& a_extraEdges,
869  VecInt2d& a_extraPoints)
870 {
871  // a_p = 5
872  const CellData& last = chain.back(); // [4, 3, 0] => [5, 3, 0, None, 4, None]
873  Edge edge = {a_p, last.m_nextPoint, last.m_cell, XM_NONE, last.m_priorPoint, XM_NONE};
874 
875  if (chain.size() > 2)
876  {
877  const CellData& first = chain[0]; // [m_priorPoint=1, m_nextPoint=7, face=6]
878  a_extraEdges.push_back({first.m_cell, last.m_cell, cost}); // [6, 0, cost]
879  // if this splits, it will produce quads:
880  // for first.m_cell: { a_p, first.m_nextPoint, first.m_priorPoint, p' }
881  // for last.m_cell: { last.m_nextPoint, last.m_priorPoint, a_p, p' }
882  //
883  // last.m_nextP \ / first.m_priorP
884  // \ p' /
885  // last.m_cell \ / first.m_cell
886  // last.m_priorP ---------- a_p -------- first.m_nextP
887  a_extraPoints.push_back(
888  {a_p, first.m_nextPoint, first.m_priorPoint, last.m_nextPoint, last.m_priorPoint});
889  // compute p'
890  Pt3d px;
891  int nTris = (int)chain.size();
892  const CellData& tri = chain[nTris >> 1];
893  const Pt3d& p0 = m_points[a_p];
894  const Pt3d& p1 = m_points[tri.m_priorPoint];
895  if (nTris & 0x1)
896  {
897  // There are an odd number of faces, so use the centroid of the
898  // middle triangle.
899  const Pt3d& p2 = m_points[tri.m_nextPoint];
900  px = (p0 + p1 + p2) / 3.0;
901  }
902  else
903  {
904  // There are an even number of faces, so use the middle triangle's
905  // incoming edge.
906  px = (p0 + p1) / 2.0;
907  }
908  m_splitPoints.push_back(px);
909  }
910  return edge;
911 
912 } // MeQuadBlossomImpl::GetBoundaryEdges

965 
966  int numPoints = a_ugrid->PointCount();
967  VecVecAdjPointMidpoint edgeMidsides(numPoints);
968  for (int pointIdx = 0; pointIdx < numPoints; ++pointIdx)
969  {
970  const Pt3d p0 = points[pointIdx];
971  VecInt adjPoints;
972  GetPointIdxsAttachedByEdge(a_ugrid, pointIdx, adjPoints);
973  VecAdjPointMidpoint adjPoints2;
974  for (auto otherIdx : adjPoints)
975  {
976  if (otherIdx > pointIdx)
977  {
978  const Pt3d& p1 = points[otherIdx];
979  Pt3d midPoint = (p0 + p1) * 0.5;
980  int midIdx = (int)points.size();
981  points.push_back(midPoint);
982  adjPoints2.push_back({ otherIdx, midIdx });
983  }
984  }
985  // sort adjPoint2 by element 0
986  std::sort(adjPoints2.begin(), adjPoints2.end());
987  edgeMidsides[pointIdx] = adjPoints2;
988  }
989 
990  // also get centroids and add to new points
991  // track start of new centroid starts
992  int numCells = a_ugrid->GetCellCount();
993  for (int cellIdx = 0; cellIdx < numCells; ++cellIdx)
994  {
995  // TODO: Use new UGrid centroid code in xmsgrid
996  VecInt cellPointIdxs = a_ugrid->GetCellPoints(cellIdx);
997  VecPt3d cellPoints = a_ugrid->GetPointsLocations(cellPointIdxs);
998  Pt3d centroid = gmComputeCentroid(cellPoints);
999  int centroidIdx = (int)points.size();
1000  points.push_back(centroid);
1001  int numCellPoints = (int)cellPointIdxs.size();
1002  int lastIdx = cellPointIdxs.back();
1003  int midsideLast = FindMidpoint(edgeMidsides, cellPointIdxs.front(), lastIdx);
1004  for (size_t i = 0; i < numCellPoints; ++i)
1005  {
1006  int currentIdx = cellPointIdxs[i];
1007  int nextIdx = cellPointIdxs[(i + 1) % numCellPoints];
1008  int midside = FindMidpoint(edgeMidsides, currentIdx, nextIdx);
1009 
1013 
1014  cells.push_back(XMU_QUAD);
1015  cells.push_back(4);
1016  cells.push_back(currentIdx);
1017  cells.push_back(midside);
1018  cells.push_back(centroidIdx);
1019  cells.push_back(midsideLast);
1020 
1021  midsideLast = midside;
1022  }
1023  }
1024
