310 {
311  // Find the lower left most point
312  double minX(XM_DBL_HIGHEST), minY(XM_DBL_HIGHEST);
313  size_t minIdx = 0;

327  minY = a_outPoly[i].y;
328  minIdx = i;
329  }
330  }
331  }
