322 void MePolyMesherImpl::SortPoly(VecPt3d& a_outPoly)
323 {
324  // Find the lower left most point
325  double minX(XM_DBL_HIGHEST), minY(XM_DBL_HIGHEST);
326  size_t minIdx = 0;
327  for (size_t i = 0; i < a_outPoly.size(); ++i)
328  {
329  if (a_outPoly[i].x < minX)
330  {
331  minX = a_outPoly[i].x;
332  minY = a_outPoly[i].y;
333  minIdx = i;
334  }
335  else if (EQ_TOL(a_outPoly[i].x, minX, m_xyTol))
336  {
337  if (a_outPoly[i].y < minY)
338  { // X is same as min. Check y
339  minX = a_outPoly[i].x;
340  minY = a_outPoly[i].y;
341  minIdx = i;
342  }
343  }
344  }
345  std::rotate(a_outPoly.begin(), a_outPoly.begin() + minIdx, a_outPoly.end());

366 bool MePolyMesherImpl::MeshFromInputs(VecPt3d& a_points, VecInt& a_triangles, VecInt& a_cells)
367 {
368  try
369  {
370  GenerateMeshPts();
371  if (m_cells.empty())
372  { // paving
373  while (m_tin->Triangles().empty())
374  {
375  ProcessBoundaryPtsFlaggedToRemove();
376  Triangulate();
377  FindAllPolyPointIdxs();
378  AutoFixFourTrianglePts();
379  AddBreaklines();
380  DeleteTrianglesOutsidePolys();
381  Relax();
382  }
383  // Re-add breaklines and delete outer polys because relaxing can swap edges
384  AddBreaklines();
385  DeleteTrianglesOutsidePolys();
386  a_triangles.swap(m_tin->Triangles());
387  a_cells.resize(0);
388  }
389  else
390  { // patch
391  a_cells.swap(m_cells);
392  a_triangles.resize(0);
393  }
394  a_points.swap(*m_points);
395  m_tin->Clear();
396  m_polyCorners.clear();
397  if (m_elev)
398  for (Pt3d& p : a_points)
399  p.z = (double)m_elev->InterpToPt(p);
400  }
401  catch (std::exception& e)
402  {
403  std::string msg = e.what();
404  meModifyMessageWithPolygonId(m_polyId, msg);
405  XM_LOG(xmlog::error, msg);
406  return false;
407  }
408  return true;
409 } // MePolyMesherImpl::MeshFromInputs
