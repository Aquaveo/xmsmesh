139 void MeMultiPolyTo2dmImpl::Write2dm(MeMultiPolyMesherIo& a_io, std::ostream& a_os,
140  int a_precision)
141 {
142  XM_ENSURE_TRUE_NO_ASSERT(a_io.m_points.size() && a_io.m_cells.size());
143 
144  a_os << "MESH2D\n";
145  std::string tempStr("E3T"), format("%5d");
146  int ival = static_cast<int>(std::log10(a_io.m_points.size())) + 1;
147  if (ival > 5)
148  {
149  std::stringstream ss;
150  ss << "%" << ival << "d";
151  format = ss.str();
152  }
153 
154  int id(0);
155  for (size_t i = 0; i < a_io.m_cells.size();)
156  {
157  ++id;
158  VecInt& cells(a_io.m_cells);
159  int celltype = cells[i + 0];
160  tempStr = "E3T";
161  if (celltype != 5) // 5 = VTK_TRIANGLE
162  tempStr = "E4Q";
163  a_os << tempStr;
164 
165  tempStr = (boost::format(format.c_str()) % id).str();
166  a_os << " " << tempStr;
167 
168  // order the points so that the one with the lowest number is first
169  int numPoints = cells[i + 1];
170  int* points = &cells[i + 2]; // points for cell start at i + 2
171  int lowPointNumber = points[0];
172  int lowPointIndex = 0;
173  for (int j = 1; j < numPoints; ++j)
174  {
175  if (points[j] < lowPointNumber)
176  {
177  lowPointNumber = points[j];
178  lowPointIndex = j;
179  }
180  }
181  for (int j = 0; j < numPoints; ++j)
182  {
183  int pointIndex = (j + lowPointIndex) % numPoints;
184  // add 1 to change from 0 based to 1 based point numbers.
185  tempStr = (boost::format(format.c_str()) % (points[pointIndex] + 1)).str();
186  a_os << " " << tempStr;
187  }
188  tempStr = (boost::format(format.c_str()) % 1).str();
189  a_os << " " << tempStr << "\n";
190 
191  i = i + 2 + numPoints;
192  }
193 
194  id = 0;
195  for (size_t i = 0; i < a_io.m_points.size(); ++i)
196  {
197  ++id;
198  tempStr = (boost::format(format.c_str()) % id).str();
199  Pt3d& p(a_io.m_points[i]);
200 
201  int test_precision = -1;
202 #ifdef TEST_PRECISION
203  test_precision = TEST_PRECISION;
204 #endif
205  a_os << "ND " << tempStr << " "
206  << STRstd(p.x, test_precision, a_precision, STR_FULLWIDTH) << " "
207  << STRstd(p.y, test_precision, a_precision, STR_FULLWIDTH) << " "
208  << STRstd(p.z, test_precision, a_precision, STR_FULLWIDTH) << "\n";
209  }
210 } // MeMultiPolyTo2dmImpl::Write2dm
