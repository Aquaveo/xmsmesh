82 static void iSortCellsForTesting(MeMultiPolyMesherIo& a_io)
83 {
84  VecInt& outCells(a_io.m_cells);
85 
86  // build array of sorted cells
87  std::vector<std::array<int, 6>> sortableCells;
88  for (size_t i = 0; i < outCells.size();)
89  {
90  std::array<int, 6> cell = {-1, -1, -1, -1, -1, -1};
91  cell[4] = outCells[i];
92  int nPts = cell[5] = outCells[i+1];
93  int minIndex(outCells[i+2]);
94  auto minItr = cell.begin();
95  for (int j=0; j<nPts; ++j)
96  {
97  int idx = outCells[i+2+j];
98  cell[j] = idx;
99  if (idx < minIndex)
100  {
101  minIndex = idx;
102  minItr = cell.begin() + j;
103  }
104  }
105  i += (size_t)nPts + 2;
106 
107  // put minimum index point first
108  std::rotate(cell.begin(), minItr, cell.begin() + nPts);
109  sortableCells.push_back(cell);
110  }
111  int cnt(0);
112  std::sort(sortableCells.begin(), sortableCells.end());
113 
114  // place cells back in output
115  for (auto& c : sortableCells)
116  {
117  outCells[cnt++] = c[4];
118  outCells[cnt++] = c[5];
119  outCells[cnt++] = c[0];
120  outCells[cnt++] = c[1];
121  outCells[cnt++] = c[2];
122  if (c[3] != -1)
123  {
124  outCells[cnt++] = c[3];
125  }
126  }
127 } // iSortCellsForTesting

198 void MeMultiPolyTo2dmImpl::Write2dm(MeMultiPolyMesherIo& a_io, std::ostream& a_os,
199  int a_precision)
200 {
201  XM_ENSURE_TRUE_NO_ASSERT(a_io.m_points.size() && a_io.m_cells.size());
202 
203  a_os << "MESH2D\n";
204  std::string tempStr("E3T"), format("%5d");
205  int ival = static_cast<int>(std::log10(a_io.m_points.size())) + 1;
206  if (ival > 5)
207  {
208  std::stringstream ss;
209  ss << "%" << ival << "d";
210  format = ss.str();
211  }
212 
213  if (m_sortCellsForTesting)
214  {
215  iSortCellsForTesting(a_io);
216  }
217 
218  int id(0);
219  for (size_t i = 0; i < a_io.m_cells.size();)
220  {
221  ++id;
222  VecInt& cells(a_io.m_cells);
223  int celltype = cells[i + 0];
224  tempStr = "E3T";
225  if (celltype != 5) // 5 = VTK_TRIANGLE
226  tempStr = "E4Q";
227  a_os << tempStr;
228 
229  tempStr = (boost::format(format.c_str()) % id).str();
230  a_os << " " << tempStr;
231 
232  // order the points so that the one with the lowest number is first
233  int numPoints = cells[i + 1];
234  int* points = &cells[i + 2]; // points for cell start at i + 2
235  int lowPointNumber = points[0];
236  int lowPointIndex = 0;
237  for (int j = 1; j < numPoints; ++j)
238  {
239  if (points[j] < lowPointNumber)
240  {
241  lowPointNumber = points[j];
242  lowPointIndex = j;
243  }
244  }
245  for (int j = 0; j < numPoints; ++j)
246  {
247  int pointIndex = (j + lowPointIndex) % numPoints;
248  // add 1 to change from 0 based to 1 based point numbers.
249  tempStr = (boost::format(format.c_str()) % (points[pointIndex] + 1)).str();
250  a_os << " " << tempStr;
251  }
252  tempStr = (boost::format(format.c_str()) % 1).str();
253  a_os << " " << tempStr << "\n";
254 
255  i = i + 2 + numPoints;
256  }
257 
258  id = 0;
259  for (size_t i = 0; i < a_io.m_points.size(); ++i)
260  {
261  ++id;
262  tempStr = (boost::format(format.c_str()) % id).str();
263  Pt3d& p(a_io.m_points[i]);
264 
265  int test_precision = -1;
266 #ifdef TEST_PRECISION
267  test_precision = TEST_PRECISION;
268 #endif
269  a_os << "ND " << tempStr << " "
270  << STRstd(p.x, test_precision, a_precision, STR_FULLWIDTH) << " "
271  << STRstd(p.y, test_precision, a_precision, STR_FULLWIDTH) << " "
272  << STRstd(p.z, test_precision, a_precision, STR_FULLWIDTH) << "\n";
273  }
274 } // MeMultiPolyTo2dmImpl::Write2dm
