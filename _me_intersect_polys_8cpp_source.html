107 static GmBstBox3d iCalcPolyEnvelope(const std::vector<size_t>& a_loop,
108  const std::vector<Pt3d>& a_pts)
109 {
110  GmBstBox3d b;
111  if (a_loop.empty() || a_pts.empty())
112  return b;
113 
114  b.min_corner() = b.max_corner() = a_pts[a_loop[0]];
115  b.min_corner().z = b.max_corner().z = 0;
116  for (size_t j = 1; j < a_loop.size(); ++j)
117  {
118  const Pt3d& p(a_pts[a_loop[j]]);
119  if (b.min_corner().x > p.x)
120  b.min_corner().x = p.x;
121  if (b.max_corner().x < p.x)
122  b.max_corner().x = p.x;
123  if (b.min_corner().y > p.y)
124  b.min_corner().y = p.y;
125  if (b.max_corner().y < p.y)
126  b.max_corner().y = p.y;
127  }
128  return b;
129 } // iCalcPolyEnvelope

385 void MeIntersectPolys::impl::InInDoIntersection()
386 {
387  while (!m_stack.empty())
388  {
389  std::list<size_t>::iterator it = m_stack.begin();
390  size_t idx = *it, idx2;
391 
392  bool processed = false;
393  std::list<size_t>::iterator it2 = it;
394  ++it2;
395  while (!processed && it2 != m_stack.end())
396  {
397  idx2 = *it2;
398  // find potentially intersecting poly
399  if (iEnvelopesOverlap(idx, idx2, m_envelopes, m_envelopes))
400  { // process the 2 polygons. If they intersect a new poly is created and
401  // the original 2 are removed from the stack. A new OUTSIDE_POLY can
402  // also be created in this process; it is moved to the output
403  processed = BoostPolyUnion(idx, idx2);
404  }
405  if (processed)
406  {
407  m_stack.erase(it);
408  m_stack.erase(it2);
409  }
410  else
411  ++it2;
412  }
413 
414  if (!processed)
415  { // move this loop to the output
416  m_stack.pop_front();
417  m_bOutPolys.push_back(m_bPolys[idx]);
418  m_bOutPolyType.push_back(MePolyOffsetter::INSIDE_POLY);
419  }
420  }
421 } // InsidePolys::impl::InInDoIntersection

580 {
581  std::vector<GmBstBox3d>& env(m_envelopes);
582  // find polys that are completely out of other poly envelopes
583  std::vector<int> overlaps(env.size(), 0);
584  for (size_t i = 0; i < env.size(); ++i)
585  {
586  if (a_delPolys.find(i) != a_delPolys.end())
587  continue;
588 
589  for (size_t j = 0; j < env.size(); j++)
590  {
591  if (i == j)
592  continue;

594  continue;

596  continue;
597 
598  if (iEnvelopesOverlap(i, j, env, env))
599  {
600  overlaps[i] = overlaps[j] = 1;
601  }
602  }
603  }
604  for (size_t i = 0; i < overlaps.size(); ++i)
605  { // anything that doesn't overlap move to the output and mark the poly
606  // as deleted
607  if (!overlaps[i] && a_delPolys.find(i) == a_delPolys.end())
608  {

610  m_out.m_loopTypes.push_back(MePolyOffsetter::INSIDE_POLY);

698  // sort the polys by envelope size
699  std::multimap<double, size_t> areaMap;
700  for (size_t i = 0; i < inPolys.size(); ++i)
701  {
702  GmBstBox3d& b(m_envelopes[inPolys[i]]);
703  double area = (b.max_corner().x - b.min_corner().x) * (b.max_corner().y - b.min_corner().y);
704  areaMap.insert(std::make_pair(area, inPolys[i]));
705  }
706  std::multimap<double, size_t>::reverse_iterator rit = areaMap.rbegin();
707  for (; rit != areaMap.rend(); ++rit)
708  {
709  m_stack.push_back(rit->second);
710  }
711 } // MeIntersectPolys::impl::InOutCalcStartingStack

773 //------------------------------------------------------------------------------
774 bool MeIntersectPolys::impl::BoostPolyUnion(size_t a_i, size_t a_j)
775 {
776  std::vector<GmBstPoly3d> out;
777  GmBstPoly3d &iPoly(m_bPolys[a_i]), &jPoly(m_bPolys[a_j]);
778  std::pair<size_t, size_t> p1(a_i, a_j), p2(a_j, a_i);
779  if (m_polyInsideOfPoly.find(p1) != m_polyInsideOfPoly.end() ||
780  m_polyInsideOfPoly.find(p2) != m_polyInsideOfPoly.end())
781  return false;
782  bg::union_(iPoly, jPoly, out);
783  if (out.size() != 1)
784  return false;
785  m_stack.push_back(m_bPolys.size());
786  m_bPolys.push_back(out[0]);
787  m_bPolys.back().inners().clear();
788  std::vector<Pt3d>& pts(m_bPolys.back().outer());
789  std::vector<size_t> loop(pts.size());
790  for (size_t i = 0; i < loop.size(); ++i)
791  loop[i] = i;
792  m_envelopes.push_back(iCalcPolyEnvelope(loop, pts));
793  // put any inner polygons into the output
794  for (size_t i = 0; i < out[0].inners().size(); ++i)
795  {
796  GmBstPoly3d innerPoly;
797  std::vector<Pt3d>& vi(out[0].inners()[i]);
798  std::vector<Pt3d>::reverse_iterator it(vi.rbegin()), iend(vi.rend());
799  for (; it != iend; ++it)
800  {
801  bg::exterior_ring(innerPoly).push_back(*it);
802  }
803  m_bOutPolys.push_back(innerPoly);
804  m_bOutPolyType.push_back(MePolyOffsetter::NEWOUT_POLY);
805  }
806  return true;
807 } // MeInsidePolys::impl::BoostPolyUnion

814 //------------------------------------------------------------------------------
815 bool MeIntersectPolys::impl::BoostPolySubtract(size_t a_i, size_t a_j)
816 {
817  std::vector<GmBstPoly3d> out;
818  GmBstPoly3d &iPoly(m_bPolys[a_i]), &jPoly(m_bPolys[a_j]);
819  if (!bg::intersects(iPoly, jPoly))
820  return false;
821  bg::difference(iPoly, jPoly, out);
822  m_inOutOpolys.erase(a_i);
823  for (size_t i = 0; i < out.size(); ++i)
824  {
825  m_inOutOpolys.insert(m_bPolys.size());
826  m_bPolys.push_back(out[i]);
827  m_bPolys.back().inners().clear();
828  std::vector<Pt3d>& pts(m_bPolys.back().outer());
829  std::vector<size_t> loop(pts.size());
830  for (size_t i = 0; i < loop.size(); ++i)
831  loop[i] = i;
832  m_envelopes.push_back(iCalcPolyEnvelope(loop, pts));
833  m_loopTypes.push_back(MePolyOffsetter::OUTSIDE_POLY);
834  }
835  return true;
836 } // MeInsidePolys::impl::BoostPolyUnion
