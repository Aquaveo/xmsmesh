143 void MeRelaxerImpl::Relax(/*MeshPolyEnum a_meshPolyEnum,*/
144  const VecInt& a_fixedPoints,
145  BSHP<TrTin> a_tin)
146 {
147  XM_ENSURE_TRUE_VOID_NO_ASSERT(a_tin);
148  XM_ENSURE_TRUE_VOID(!a_tin->TrisAdjToPts().empty());
149  XM_ENSURE_TRUE_VOID(AllTrianglesHavePositiveArea(a_tin));
150  // Set up member variables
152  // m_meshPolyEnum = a_meshPolyEnum;
153  m_tin = a_tin;
154  m_fixedPoints = &a_fixedPoints;
155 
156  // Mark fixed points to not be relaxed (breaklines)
157  m_flags.assign(m_tin->Points().size(), RELAX_RELAX);
158  for (size_t i = 0; i < a_fixedPoints.size(); ++i)
159  {
160  XM_ASSERT(a_fixedPoints[i] >= 0 && a_fixedPoints[i] < (int)m_flags.size());
161  m_flags[a_fixedPoints[i]] = 0;
162  }
163 
164  ComputeCentroids();
165 
166  // Set up iterations
167  VecPt3d& pts(m_tin->Points());
168  int numiterations(3); // default to 3 for area relaxation
169  RelaxTypeEnum relaxtype(m_relaxType);
170  if (relaxtype == RELAXTYPE_SPRING)
171  {
172  if (!m_sizer)
173  {
174  std::string msg =
175  "No size function specified with spring relaxation "
176  "method. Relaxation method has been set to AREA "
177  "relaxation.";
178  XM_LOG(xmlog::warning, msg);
179  relaxtype = RELAXTYPE_AREA;
180  }
181  else
182  {
183  if (m_pointSizes.empty())
184  {
185  SetupPointSizes();
186  }
187  SetupNeighbors();
188  XM_ASSERT(pts.size() == m_pointNeighbors.size());
189  XM_ASSERT(pts.size() == m_pointSizes.size());
190  }
191  }
192  XM_ASSERT(pts.size() == m_flags.size());
193 
194  int iteration = 0;
195 
196  // Perform relaxation and swap
197  for (int i = 0; i < numiterations; ++i)
198  {
199  ++iteration;
200  RelaxMarkedPoints(relaxtype, iteration, numiterations);
201  // delete points and triangles
202  if (!m_pointsToDelete.empty())
203  {
204  m_tin->Triangles().clear();
205  m_tin->TrisAdjToPts().clear();
206  VecPt3d& pts(m_tin->Points());
207  VecDbl& szs(m_pointSizes);
208  for (auto it = m_pointsToDelete.rbegin(); it != m_pointsToDelete.rend(); ++it)
209  {
210  int idx = *it;
211  pts[idx] = pts.back();
212  pts.pop_back();
213  szs[idx] = szs.back();
214  szs.pop_back();
215  m_flags[idx] = m_flags.back();
216  m_flags.pop_back();
217  }
218  m_pointsToDelete.clear();
219  break;
220  }
221  bool trianglesChanged(false);
222  trianglesChanged = m_tin->OptimizeTriangulation();
223 
224  if (trianglesChanged && RELAXTYPE_SPRING == relaxtype)
225  { // set up spring relax if the triangles have changed
226  SetupNeighbors();
227  }
228  }
229 
230 } // MeRelaxerImpl::Relax

266 void MeRelaxerImpl::RelaxMarkedPoints(RelaxTypeEnum a_relaxType,
267  int a_iteration,
268  int a_numiterations)
269 {
270  XM_ASSERT(a_iteration > 0 && a_numiterations > 0 && a_iteration <= a_numiterations);
271 
272  // Progress
273  double iterationPercent = a_iteration / (double)a_numiterations;
274 
275  // Relax the marked nodes
276  VecPt3d& points = m_tin->Points(); // for convenience
277  VecInt2d& trisAdjToPts = m_tin->TrisAdjToPts();
278  size_t nPoints = points.size();
279  Pt3d origlocation, newlocation;
280  for (size_t p = 0; p < nPoints; ++p)
281  {
282  if (m_flags[p])
283  {
284  origlocation = points[p];
285  // do general point relax
286  switch (a_relaxType)
287  {
288  case RELAXTYPE_AREA:
289  AreaRelax((int)p, newlocation);
290  break;
291  case RELAXTYPE_ANGLE:
292  AngleRelax((int)p, newlocation);
293  break;
294  case RELAXTYPE_SPRING:
295  SpringRelaxSinglePoint((int)p, newlocation);
296  break;
297  default:
298  XM_ASSERT(false);
299  break;
300  }
301 
302  // Change the points location and check for bad triangles
303  if (NewLocationIsValid(p, newlocation))
304  {
305  points[p] = newlocation;
306  if (m_sizer)
307  {
308  // point moved and we must update its target size when doing
309  // spring_relax
310  m_pointSizes[p] = m_sizer->SizeFromLocation(points[p]);
311  }
312  }
313  else if (RELAXTYPE_SPRING == a_relaxType)
314  { // mark point for deletion
315  m_pointsToDelete.push_back(static_cast<int>(p));
316  }
317 
318  } // if (m_flags[p])
319 
320  // Update progress
321  if (m_observer)
322  {
323  double pcntdone = iterationPercent * p / (double)nPoints;
324  m_observer->ProgressStatus(pcntdone);
325  }
326  } // for (int p = 0; p < nPoints; ++p)
327 } // MeRelaxerImpl::RelaxMarkedPoints

355 void MeRelaxerImpl::AngleRelax(int a_point, Pt3d& a_newLocation)
356 {
357  int id;
358  double targetangle, mag, a, c, d;
359  double da, db, dx, dy, sx, sy, radius;
360  Pt2d sum, pt, p1, p2, p3, centroid;
361  int point1, point2;
362 
363  // get the target angle
364  const VecInt& adjTris = m_tin->TrisAdjToPts()[a_point];
365  VecPt3d& points = m_tin->Points();
366  size_t num = adjTris.size();
367  XM_ASSERT(num > 0);
368 
369  targetangle = 2.0 * XM_PI / (double)num;
370  // loop through adjacent elems and find the
371  // circle with the two adjacent points and
372  // the new point
373  sum.x = sum.y = 0.0;
374  for (size_t i = 0; i < adjTris.size(); ++i)
375  {
376  int tri = adjTris[i];
377  id = m_tin->LocalIndex(tri, a_point);
378  point1 = m_tin->GlobalIndex(tri, trIncrementIndex(id));
379  point2 = m_tin->GlobalIndex(tri, trDecrementIndex(id));
380  // set up points
381  pt.x = points[a_point].x;
382  pt.y = points[a_point].y;
383  p1.x = points[point1].x;
384  p1.y = points[point1].y;
385  p2.x = points[point2].x;
386  p2.y = points[point2].y;
387  // find length of line bisecting p1 and p2
388  dx = p1.x - p2.x;
389  dy = p1.y - p2.y;
390  c = sqrt(sqr(dx) + sqr(dy));
391  sx = p1.x + dx / 2.0;
392  sy = p1.y + dy / 2.0;
393  // compute length of triangle sides
394  a = (c / 2.0) / sin(targetangle / 2.0);
395  d = a * cos(targetangle / 2.0);
396  radius = (a * a) / (2.0 * d);
397  // find candidate point p3 and circle centroid
398  dx /= c;
399  dy /= c;
400  p3.x = sx + d * dy;
401  p3.y = sy - d * dx;
402  da = sqr(p3.x - pt.x) + sqr(p3.y - pt.y);
403  p3.x = sx - d * dy;
404  p3.y = sy + d * dx;
405  db = sqr(p3.x - pt.x) + sqr(p3.y - pt.y);
406  if (da < db)
407  {
408  centroid.x = sx + (d - radius) * dy;
409  centroid.y = sy - (d - radius) * dx;
410  }
411  else
412  {
413  centroid.x = sx - (d - radius) * dy;
414  centroid.y = sy + (d - radius) * dx;
415  }
416  // find p3 along circle, radius away from centroid
417  dx = pt.x - centroid.x;
418  dy = pt.y - centroid.y;
419  mag = sqrt(dx * dx + dy * dy);
420  dx /= mag;
421  dy /= mag;
422  // find new point
423  p3.x = centroid.x + radius * dx;
424  p3.y = centroid.y + radius * dy;
425  // sum new point's x and y
426  sum.x += p3.x;
427  sum.y += p3.y;
428  }
429  a_newLocation.x = (sum.x / (double)num);
430  a_newLocation.y = (sum.y / (double)num);
431 } // MeRelaxerImpl::AngleRelax

438 void MeRelaxerImpl::SpringRelaxSinglePoint(int a_point, Pt3d& a_newLocation)
439 {
440  double fx(0.0);
441  double fy(0.0);
442 
443  double springStiffness(1.0);
444  double stepSize(0.1);
445  VecPt3d& points = m_tin->Points();
446  VecInt& neighbors(m_pointNeighbors[a_point]);
447  double numPtsFactor(1.7 / (double)neighbors.size());
448  Pt3d& p0(points[a_point]);
449  double size0(m_pointSizes[a_point]);
450  for (size_t i = 0; i < neighbors.size(); ++i)
451  {
452  int neighborIdx = neighbors[i];
453  Pt3d& p1(points[neighborIdx]);
454  double size1(m_pointSizes[neighborIdx]);
455  Pt3d vec = gmCreateVector(p0, p1);
456  double dist = Mdist(p0.x, p0.y, p1.x, p1.y);
457  double targetDist = 0.5 * (size0 + size1);
458  double force(springStiffness * (dist - targetDist));
459  double _fx = force * vec.x / dist;
460  double _fy = force * vec.y / dist;
461  fx += _fx;
462  fy += _fy;
463  }
464  fx *= numPtsFactor;
465  fy *= numPtsFactor;
466  a_newLocation.x = p0.x + fx;
467  a_newLocation.y = p0.y + fy;
468  a_newLocation.z = p0.z;
469 } // MeRelaxerImpl::SpringRelax

474 void MeRelaxerImpl::SetupNeighbors()
475 {
476  m_pointNeighbors.clear();
477  VecPt3d& points = m_tin->Points();
478  size_t nPts(points.size());
479  if (m_flags.empty())
480  {
481  m_flags.assign(nPts, RELAX_RELAX);
482  }
483  m_pointNeighbors.assign(nPts, VecInt());
484  VecInt& tris = m_tin->Triangles();
485  VecInt2d& adjTris2d = m_tin->TrisAdjToPts();
486  for (size_t i = 0; i < points.size(); ++i)
487  {
488  if (m_flags[i] == 0) // We don't relax fixed points, so we don't need its neighbors.
489  {
490  continue;
491  }
492  const VecInt& adjTris = adjTris2d[i];
493  size_t numTri = adjTris.size();
494  // find the other points that make up the edges that a_point is connected to
495  std::set<int> setPoints;
496  for (size_t j = 0; j < numTri; ++j)
497  {
498  int idx = adjTris[j] * 3;
499  for (int t = 0; t < 3; ++t)
500  {
501  int ptIdx = tris[idx + t];
502  XM_ASSERT(ptIdx < nPts);
503  setPoints.insert(tris[idx + t]);
504  }
505  }
506  setPoints.erase(static_cast<int>(i));
507  m_pointNeighbors[i] = VecInt(setPoints.begin(), setPoints.end());
508  }
509 } // MeRelaxerImpl::SetupNeighbors()
