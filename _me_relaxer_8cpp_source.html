354 bool MeRelaxerImpl::AngleRelax(int a_point, Pt3d& a_newLocation)
355 {
356  int id;
357  double targetangle, mag, a, c, d;
358  double da, db, dx, dy, sx, sy, radius;
359  Pt2d sum, pt, p1, p2, p3, centroid;
360  int point1, point2;
361 
362  // get the target angle
363  const VecInt& adjTris = m_tin->TrisAdjToPts()[a_point];
364  VecPt3d& points = m_tin->Points();
365  size_t num = adjTris.size();
366  XM_ASSERT(num > 0);
367 
368  targetangle = 2.0 * XM_PI / (double)num;
369  // loop through adjacent elems and find the
370  // circle with the two adjacent points and
371  // the new point
372  sum.x = sum.y = 0.0;
373  for (size_t i = 0; i < adjTris.size(); ++i)
374  {
375  int tri = adjTris[i];
376  id = m_tin->LocalIndex(tri, a_point);
377  point1 = m_tin->GlobalIndex(tri, trIncrementIndex(id));
378  point2 = m_tin->GlobalIndex(tri, trDecrementIndex(id));
379  // set up points
380  pt.x = points[a_point].x;
381  pt.y = points[a_point].y;
382  p1.x = points[point1].x;
383  p1.y = points[point1].y;
384  p2.x = points[point2].x;
385  p2.y = points[point2].y;
386  // find length of line bisecting p1 and p2
387  dx = p1.x - p2.x;
388  dy = p1.y - p2.y;
389  c = sqrt(sqr(dx) + sqr(dy));
390  sx = p1.x + dx / 2.0;
391  sy = p1.y + dy / 2.0;
392  // compute length of triangle sides
393  a = (c / 2.0) / sin(targetangle / 2.0);
394  d = a * cos(targetangle / 2.0);
395  radius = (a * a) / (2.0 * d);
396  // find candidate point p3 and circle centroid
397  dx /= c;
398  dy /= c;
399  p3.x = sx + d * dy;
400  p3.y = sy - d * dx;
401  da = sqr(p3.x - pt.x) + sqr(p3.y - pt.y);
402  p3.x = sx - d * dy;
403  p3.y = sy + d * dx;
404  db = sqr(p3.x - pt.x) + sqr(p3.y - pt.y);
405  if (da < db)
406  {
407  centroid.x = sx + (d - radius) * dy;
408  centroid.y = sy - (d - radius) * dx;
409  }
410  else
411  {
412  centroid.x = sx - (d - radius) * dy;
413  centroid.y = sy + (d - radius) * dx;
414  }
415  // find p3 along circle, radius away from centroid
416  dx = pt.x - centroid.x;
417  dy = pt.y - centroid.y;
418  mag = sqrt(dx * dx + dy * dy);
419  dx /= mag;
420  dy /= mag;
421  // find new point
422  p3.x = centroid.x + radius * dx;
423  p3.y = centroid.y + radius * dy;
424  // sum new point's x and y
425  sum.x += p3.x;
426  sum.y += p3.y;
427  }
428  a_newLocation.x = (sum.x / (double)num);
429  a_newLocation.y = (sum.y / (double)num);
430  return true;
431 } // MeRelaxerImpl::AngleRelax

440 bool MeRelaxerImpl::RelaxOnBoundary(int a_point, Pt3d& a_newLocation, int a_relaxType)
441 {
442  double angle, minangle, pcnt, dxp, dyp, dxn, dyn, dx1, dx2, dy1, dy2, totdist, dist1, dist2;
443 
444  VecPt3d& points = m_tin->Points();
445 
446  minangle = m_slideangle * (XM_PI / 180.0);
447  int nextPoint(-1), prevPoint(-1);
448 
449  if (m_flags[a_point] == RELAX_MATSLIDE)
450  {
452  XM_ASSERT(false);
453  }
454  else
455  {
456  prevPoint = m_tin->PreviousBoundaryPoint(a_point);
457  nextPoint = m_tin->NextBoundaryPoint(a_point);
458  }
459 
460  if (prevPoint != -1 && nextPoint != -1)
461  {
462  dxp = points[prevPoint].x - points[a_point].x;
463  dyp = points[prevPoint].y - points[a_point].y;
464  dxn = points[nextPoint].x - points[a_point].x;
465  dyn = points[nextPoint].y - points[a_point].y;
466  angle = gmAngleBetween2DVectors(dxp, dyp, dxn, dyn);
467  angle -= XM_PI;
468  if (fabs(angle) > minangle)
469  {
470  return false;
471  }
472 
473  dx1 = points[nextPoint].x - points[a_point].x;
474  dy1 = points[nextPoint].y - points[a_point].y;
475  dx2 = points[prevPoint].x - points[a_point].x;
476  dy2 = points[prevPoint].y - points[a_point].y;
477  dist1 = sqrt(sqr(dx1) + sqr(dy1));
478  dist2 = sqrt(sqr(dx2) + sqr(dy2));
479  totdist = dist1 + dist2;
480 
481  // size func based relax
482  switch (a_relaxType)
483  {
484  case RELAXTYPE_DENSITY:
485  case RELAXTYPE_CIRCLE:
487  XM_ASSERT(false);
488  break;
489  case RELAXTYPE_AREA:
490  case RELAXTYPE_ANGLE:
491  if (dist1 > dist2)
492  {
493  pcnt = (0.5 * totdist - dist2) / dist1;
494  a_newLocation.x = pcnt * dx1 + points[a_point].x;
495  a_newLocation.y = pcnt * dy1 + points[a_point].y;
496  }
497  else if (dist2 > dist1)
498  {
499  pcnt = (0.5 * totdist - dist1) / dist2;
500  a_newLocation.x = pcnt * dx2 + points[a_point].x;
501  a_newLocation.y = pcnt * dy2 + points[a_point].y;
502  }
503  break;
504  default:
505  XM_ASSERT(0);
506  return false;
507  break;
508  }
509  }
510 
511  return true;
512 } // MeRelaxerImpl::RelaxOnBoundary
