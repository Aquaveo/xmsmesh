92 bool SmoothIo::CalcMaxSize(double a_length, float a_smoothVal, double& a_maxSize)
93 {
94  if (m_checkMinSize)
95  {
96  double sizeOneAway = a_length / (m_scaleFactor * a_smoothVal);
97  double negLogCheck = sizeOneAway * (m_percentGrowth - 1.0) + 1.0;
98  XM_ENSURE_TRUE(negLogCheck > 0.0, false);
99  double numElems = log(negLogCheck) / m_logPercentGrowth;
100  a_maxSize = pow(m_percentGrowth, numElems) * a_smoothVal;
101  }
102  else
103  {
104  a_maxSize = (double)a_smoothVal + a_length * m_maxSize;
105  }
106  return true;
107 } // SmoothIo::CalcMaxSize

134 static void meiDoSmooth(SmoothIo& a_)
135 {
136  if (a_.m_ptsFlag.empty())
137  a_.m_ptsFlag.resize(a_.m_tin->NumPoints(), true);
138  XM_ENSURE_TRUE(a_.m_ptsFlag.size() == (size_t)a_.m_tin->NumPoints());
139 
140  DynBitset ptsFlag(a_.m_tin->NumPoints(), false);
141 
142  VecFlt& smoothSize(*a_.m_smoothSize);
143  const VecFlt& sz(*a_.m_sizes);
144  // set the smoothed size equal to the incoming size
145  smoothSize = sz;
146  // sort points based on size
147  std::multimap<float, int> mapSizeIdx;
148  std::vector<std::multimap<float, int>::iterator> vecIt;
149  float val(1.0);
150  // if we are anchoring to the max size then make the map key the negative
151  // of the size
152  if (1 == a_.m_anchorType)
153  val = -1.0;
154  for (int i = 0; i < (int)sz.size(); ++i)
155  {
156  vecIt.push_back(mapSizeIdx.insert(std::make_pair(val * sz[i], i)));
157  }
158 
159  // get references to the Tin
160  const VecPt3d& pts(a_.m_tin->Points());
161  const VecInt2d& trisAdjToPts(a_.m_tin->TrisAdjToPts());
162  const VecInt& tris(a_.m_tin->Triangles());
163 
164  // iterate through the points
165  auto it = mapSizeIdx.begin();
166  auto endIt = mapSizeIdx.end();
167  for (; it != endIt; ++it)
168  {
169  int i = it->second; // index of the point
170  if (!a_.m_ptsFlag[i])
171  continue;
172  if (a_.m_checkMinSize && (double)smoothSize[i] < a_.m_minSize)
173  {
174  smoothSize[i] = (float)a_.m_minSize;
175  }
176  const VecInt adjTris(trisAdjToPts[i]); // triangles adjacent to the point
177  for (const auto& t : adjTris)
178  {
179  int tIdx = t * 3;
180  int triPts[3];
181  // points in the adjacent triangle
182  triPts[0] = tris[tIdx];
183  triPts[1] = tris[tIdx + 1];
184  triPts[2] = tris[tIdx + 2];
185  for (int t1 = 0; t1 < 3; ++t1)
186  {
187  int ix = triPts[t1]; // index of triangle point
188  // make sure this is not the vertex in the "i" loop
189  if (ix == i)
190  continue;
191  // make sure we have not already adjusted this point
192  if (ptsFlag[ix])
193  continue;
194 
195  const Pt3d &p0(pts[i]), &p1(pts[ix]);
196  // calculate what the min or max size can be based on how close
197  // the elements are
198  double length = Mdist(p0.x, p0.y, p1.x, p1.y);
199  double maxSize(0);
200  XM_ENSURE_TRUE(a_.CalcMaxSize(length, smoothSize[i], maxSize));
201 
202  switch (a_.m_anchorType)
203  {
204  case 0: // anchor to the min size
205  {
206  if (a_.m_checkMinSize)
207  XM_ENSURE_TRUE(maxSize > a_.m_minSize);
208  XM_ENSURE_TRUE(maxSize >= (double)smoothSize[i]);
209  if (maxSize < (double)smoothSize[ix])
210  {
211  smoothSize[ix] = (float)maxSize;
212  ptsFlag.set(ix, true);
213  }
214  else if (a_.m_checkMinSize && (double)smoothSize[ix] < a_.m_minSize)
215  {
216  smoothSize[ix] = (float)a_.m_minSize;
217  ptsFlag.set(ix, true);
218  }
219  }
220  break;
221  case 1: // anchor to the max size
222  {
223  double minSize = a_.CalcMinSize(length, smoothSize[i], maxSize);
224  // double minSize = smoothSize[i] - (maxSize - smoothSize[i]);
225  XM_ENSURE_TRUE(minSize < smoothSize[i]);
226  if (minSize > (double)smoothSize[ix])
227  {
228  smoothSize[ix] = (float)minSize;
229  ptsFlag.set(ix, true);
230  }
231  else if (a_.m_checkMinSize && (double)smoothSize[ix] < a_.m_minSize)
232  {
233  smoothSize[ix] = (float)a_.m_minSize;
234  ptsFlag.set(ix, true);
235  }
236  }
237  break;
238  default:
239  XM_ASSERT(0);
240  break;
241  }
242  // resort the point if it changed size
243  if (ptsFlag[ix])
244  {
245  ptsFlag.set(ix, false);
246  mapSizeIdx.erase(vecIt[ix]);
247  vecIt[ix] = mapSizeIdx.insert(std::make_pair(val * smoothSize[ix], ix));
248  }
249  }
250  }
251  }
252 } // meiDoSmooth

265 void meSizeFunctionFromDepth(const VecDbl& a_depths,
266  VecDbl& a_size,
267  double a_minSize,
268  double a_maxSize)
269 {
270  XM_ENSURE_TRUE(a_minSize > 0);
271  XM_ENSURE_TRUE(a_maxSize > a_minSize);
272  double minD = *std::min_element(a_depths.begin(), a_depths.end());
273  double maxD = *std::max_element(a_depths.begin(), a_depths.end());
274  double dDiff = maxD - minD;
275  XM_ENSURE_TRUE(dDiff > 0);
276  double sDiff = a_maxSize - a_minSize;
277  XM_ENSURE_TRUE(sDiff > 0);
278 
279  a_size.assign(a_depths.size(), a_minSize);
280  int i(0);
281  for (auto& d : a_size)
282  {
283  d += ((a_depths[i++] - minD) / dDiff) * sDiff;
284  }
285 } // meSizeFunctionFromDepth
