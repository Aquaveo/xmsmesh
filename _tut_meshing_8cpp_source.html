63 bool tutReadMeshIoFromFile(const std::string& a_fname, MeMultiPolyMesherIo& a_io)
64 {
65  a_io = MeMultiPolyMesherIo();
66  std::fstream os(a_fname, std::fstream::in);
67  if (!os.is_open())
68  return false;
69 
70  std::string card;
71  size_t numpts;
72  Pt3d pt;
73  MePolyInput* p(nullptr);
74  while (os.good())
75  {
76  os >> card;
77  if ("END_POLYGON" == card)
78  { // do nothing
79  p = nullptr;
80  }
81  else if ("BEGIN_POLYGON" == card)
82  {
83  a_io.m_polys.push_back(MePolyInput());
84  p = &a_io.m_polys.back();
85  }
86  else if ("OUTSIDE" == card && p)
87  {
88  os >> numpts;
89  p->m_outPoly.reserve(numpts);
90  for (size_t i = 0; i < numpts; ++i)
91  {
92  os >> pt.x >> pt.y;
93  p->m_outPoly.push_back(pt);
94  }
95 
96  VecPt3d& vPoly(p->m_outPoly);
97  double area = gmPolygonArea(&vPoly[0], vPoly.size());
98  if (area > 0.0)
99  {
100  std::reverse(vPoly.begin(), vPoly.end());
101  area = gmPolygonArea(&vPoly[0], vPoly.size());
102  }
103  }
104  else if ("INSIDE" == card && p)
105  {
106  p->m_insidePolys.push_back(VecPt3d());
107  std::vector<Pt3d>& in(p->m_insidePolys.back());
108  os >> numpts;
109  in.reserve(numpts);
110  for (size_t i = 0; i < numpts; ++i)
111  {
112  os >> pt.x >> pt.y;
113  in.push_back(pt);
114  }
115 
116  VecPt3d& vPoly(in);
117  double area = gmPolygonArea(&vPoly[0], vPoly.size());
118  if (area < 0.0)
119  {
120  std::reverse(vPoly.begin(), vPoly.end());
121  area = gmPolygonArea(&vPoly[0], vPoly.size());
122  }
123  }
124  else if ("BIAS" == card && p)
125  {
126  os >> p->m_bias;
127  }
128  else if (("SIZE_FUNCTION" == card || "ELEVATION_FUNCTION" == card) && p)
129  {
130  BSHP<InterpBase> interp;
131  std::string interpType;
132  os >> interpType; // LINEAR or IDW
133  if ("LINEAR" == interpType)
134  {
135  interp = InterpLinear::New();
136  }
137  else if ("IDW" == interpType)
138  {
139  interp = InterpIdw::New();
140  }
141 
142  os >> numpts;
143  BSHP<VecPt3d> vpts(new VecPt3d());
144  VecPt3d& pts(*vpts);
145  for (int i = 0; i < (int)numpts; ++i)
146  {
147  os >> pt.x >> pt.y >> pt.z;
148  pts.push_back(pt);
149  }
150 
151  if (interp)
152  {
153  BSHP<VecInt> tris(new VecInt());
154  TrTriangulatorPoints tri(pts, *tris);
155  tri.Triangulate();
156  interp->SetPtsTris(vpts, tris);
157  if ("SIZE_FUNCTION" == card)
158  p->m_sizeFunction = interp;
159  else
160  p->m_elevFunction = interp;
161  }
162  }
163  else if ("CONST_SIZE_FUNCTION" == card && p)
164  {
165  os >> p->m_constSizeFunction >> p->m_constSizeBias;
166  }
167  else if ("PATCH_CORNERS" == card && p)
168  {
169  p->m_polyCorners.assign(3, -1);
170  os >> p->m_polyCorners[0] >> p->m_polyCorners[1] >> p->m_polyCorners[2];
171  }
172  else if ("CHECK_TOPOLOGY" == card)
173  {
174  a_io.m_checkTopology = true;
175  }
176  else if ("RETURN_CELL_POLYGONS" == card)
177  {
178  a_io.m_returnCellPolygons = true;
179  }
180  else if ("REFINE_POINTS" == card)
181  {
182  os >> numpts;
183  a_io.m_refPts.reserve(numpts);
184  MeRefinePoint rpt(Pt3d(), -1, false);
185  for (int i = 0; i < (int)numpts; ++i)
186  {
187  os >> rpt.m_pt.x >> rpt.m_pt.y >> rpt.m_size >> rpt.m_createMeshPoint;
188  a_io.m_refPts.push_back(rpt);
189  }
190  }
191  card = "";
192  }
193  return true;
194 } // tutReadMeshIoFromFile

203 {
204  a_outside.resize(0);
