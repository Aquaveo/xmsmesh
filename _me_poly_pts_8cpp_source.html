212 static double iCalcIntersectionT(const Pt3d& a_p0, const Pt3d& a_p1, Pt3d& a_iPt, double a_xyTol)
213 {
214  double tval(-1);
215  double denomX = a_p1.x - a_p0.x;
216  double denomY = a_p1.y - a_p0.y;
217  if (fabs(denomX) > fabs(denomY))
218  {
219  tval = (a_iPt.x - a_p0.x) / denomX;
220  }
221  else
222  {
223  tval = (a_iPt.y - a_p0.y) / denomY;
224  }
225  if (gmEqualPointsXY(a_p0, a_iPt, a_xyTol) || (tval < 0 && EQ_TOL(tval, 0, T_TOL)))
226  {
227  a_iPt = a_p0;
228  tval = 0;
229  }
230  if (gmEqualPointsXY(a_p1, a_iPt, a_xyTol) || (tval > 1 && EQ_TOL(tval, 1, T_TOL)))
231  {
232  a_iPt = a_p1;
233  tval = 1;
234  }

349 static void iCheckIntersection(size_t a_i,
350  size_t a_j,
351  const std::vector<size_t>& a_iSeg,
352  const std::vector<size_t>& a_jSeg,
353  BSHP<std::vector<Pt3d>>& a_shptrPts,

357 {

359  if (iCheckSharedPtIdx(a_i, a_j, a_iSeg, a_jSeg, spts, a_xyTol, a_mapSegCross))
360  return;
361 
362  // first segment
363  size_t nexti = iNextSegIdx(a_i, a_iSeg);
364  Pt3d &s1p0(spts[a_iSeg[a_i]]), &s1p1(spts[a_iSeg[nexti]]);
365  // second segment
366  size_t nextj = iNextSegIdx(a_j, a_jSeg);
367  Pt3d &s2p0(spts[a_jSeg[a_j]]), &s2p1(spts[a_jSeg[nextj]]);
368

370  bool intersect = iIntersectTwoSegments(s1p0, s1p1, s2p0, s2p1, a_xyTol, pts);
371  if (intersect)
372  {
373  for (size_t i = 0; i < pts.size(); ++i)
374  {
375  // calculate the "t" value for the intersection on the first segment
376  double t = iCalcIntersectionT(s1p0, s1p1, pts[i], a_xyTol);
377  double t2 = iCalcIntersectionT(s2p0, s2p1, pts[i], a_xyTol);
378  // if t = 1 then we want this intersection to go on the second segment
379  // with the t value on the second segment so that intersections on that
381  if (1 == t)
382  {
383  t = iCalcIntersectionT(s2p0, s2p1, pts[i], a_xyTol);
384  iAddIntersection(a_j, a_jSeg, a_iSeg[nexti], t, a_mapSegCross);
385  }
386  // > 0 is here because at some point in our looping we will consider
387  // the same location for both t = 0 and 1 if an endpoint of a segment
388  // touches another segment. To keep from adding it twice we only add
389  // when t > 0. See testCase2 as an example.
390  else if (t > 0)
391  {
392  if (1 == t2 || 0 == t2)
393  {
394  if (1 == t2)
395  iAddIntersection(a_i, a_iSeg, a_jSeg[nextj], t, a_mapSegCross);
396  // ignore 0 because we will hit this point when t2 is 1
397  }
398  else
399  {
400  size_t ptIdx;
401  if (!a_ps)
402  {
403  ptIdx = (int)spts.size();
404  spts.push_back(pts[i]);
405  }
406  else
407  {
408  int idx;
409  a_ps->AddPtToVectorIfUnique(pts[i], a_xyTol, idx);
410  ptIdx = (size_t)idx;
411  }
412  iAddIntersection(a_i, a_iSeg, ptIdx, t, a_mapSegCross);
413  iAddIntersection(a_j, a_jSeg, ptIdx, t2, a_mapSegCross);
414  }
415  }
416  if (t < 0 || t > 1)
417  {
418  XM_ASSERT(0);
419  }
420  }
421  }
